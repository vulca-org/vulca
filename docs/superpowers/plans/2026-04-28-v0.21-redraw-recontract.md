# Vulca v0.21 Redraw Recontract Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Replace v0.20's full-canvas sparse-layer redraw with an explicit provider edit contract plus crop/per-instance redraw strategies that preserve layer geometry and fail loudly on known quality regressions.

**Architecture:** v0.21 splits redraw into three concerns: provider edit capabilities, layer geometry/strategy selection, and local quality gates. Dense layers use the provider's supported full-canvas edit path; sparse single-object layers redraw a padded bbox crop; sparse multi-instance layers redraw bounded per-component crops and stitch them back into the original canvas. The OpenAI provider stops relying on implicit `/images/edits` behavior by declaring whether a model needs a mask, supports unmasked edits, and supports high-fidelity knobs.

**Tech Stack:** Python 3.11, Pillow, numpy, pytest, existing Vulca provider registry, OpenAI image provider, MCP `layers_redraw`.

**Context:** v0.20 fixed one bug class, but it did not fully solve the design problem. The current branch shows:
- Sparse layers are detected with live alpha and routed to `inpaint_with_mask`.
- `model`, `quality`, `input_fidelity`, and `output_format` now reach the OpenAI wire form.
- `gpt-image-2` rejected maskless `/v1/images/edits` in the 2026-04-27 dogfood run.
- The route still operates on the full 4032x3024 canvas, then downscales to an API canvas and upsamples back. That loses spatial evidence for tiny subjects and does not distinguish single-object vs multi-instance layers.

```python
from __future__ import annotations

from dataclasses import dataclass
from enum import Enum

import numpy as np
from PIL import Image


class RedrawRoute(str, Enum):
    DENSE_FULL_CANVAS = "dense_full_canvas"
    SPARSE_BBOX_CROP = "sparse_bbox_crop"
    SPARSE_PER_INSTANCE = "sparse_per_instance"


@dataclass(frozen=True)
class CropBox:
    x: int
    y: int
    w: int
    h: int


@dataclass(frozen=True)
class AlphaGeometry:
    area_pct: float
    bbox_fill: float
    component_count: int
    bbox: CropBox
    components: tuple[CropBox, ...]


@dataclass(frozen=True)
class RedrawPlan:
    route: RedrawRoute
    crop_boxes: tuple[CropBox, ...]
    warnings: tuple[str, ...] = ()


def analyze_alpha_geometry(alpha: Image.Image) -> AlphaGeometry:
    arr = np.array(alpha.convert("L")) > 0
    visible = int(arr.sum())
    total = int(arr.size)
    if visible == 0 or total == 0:
        return AlphaGeometry(0.0, 0.0, 0, CropBox(0, 0, 0, 0), ())

    ys, xs = np.where(arr)
    bbox = CropBox(int(xs.min()), int(ys.min()), int(xs.max() - xs.min() + 1), int(ys.max() - ys.min() + 1))
    bbox_fill = visible / float(bbox.w * bbox.h)

    components = _connected_component_boxes(arr)
    return AlphaGeometry(
        area_pct=100.0 * visible / total,
        bbox_fill=bbox_fill,
        component_count=len(components),
        bbox=bbox,
        components=tuple(components),
    )


def choose_redraw_route(
    geom: AlphaGeometry,
    *,
    area_threshold: float = 5.0,
    bbox_fill_threshold: float = 0.5,
) -> RedrawPlan:
    if geom.component_count == 0:
        return RedrawPlan(RedrawRoute.SPARSE_BBOX_CROP, ())
    if geom.component_count > 1 and geom.bbox_fill < bbox_fill_threshold:
        return RedrawPlan(RedrawRoute.SPARSE_PER_INSTANCE, tuple(_pad_box(c, geom.bbox, 0.35) for c in geom.components))
    if geom.area_pct < area_threshold or geom.bbox_fill < bbox_fill_threshold:
        return RedrawPlan(RedrawRoute.SPARSE_BBOX_CROP, (_pad_box(geom.bbox, geom.bbox, 0.35),))
    return RedrawPlan(RedrawRoute.DENSE_FULL_CANVAS, (geom.bbox,))
```

Add the private helpers in the same file:

```python
def _connected_component_boxes(mask: np.ndarray) -> list[CropBox]:
    h, w = mask.shape
    seen = np.zeros_like(mask, dtype=bool)
    boxes: list[CropBox] = []
    for y in range(h):
        for x in range(w):
            if not mask[y, x] or seen[y, x]:
                continue
            stack = [(x, y)]
            seen[y, x] = True
            xs: list[int] = []
            ys: list[int] = []
            while stack:
                cx, cy = stack.pop()
                xs.append(cx)
                ys.append(cy)
                for nx, ny in ((cx - 1, cy), (cx + 1, cy), (cx, cy - 1), (cx, cy + 1)):
                    if 0 <= nx < w and 0 <= ny < h and mask[ny, nx] and not seen[ny, nx]:
                        seen[ny, nx] = True
                        stack.append((nx, ny))
            boxes.append(CropBox(min(xs), min(ys), max(xs) - min(xs) + 1, max(ys) - min(ys) + 1))
    return boxes


def _pad_box(box: CropBox, canvas_box: CropBox, pad_ratio: float) -> CropBox:
    pad = max(8, int(max(box.w, box.h) * pad_ratio))
    max_x = canvas_box.x + canvas_box.w
    max_y = canvas_box.y + canvas_box.h
    x0 = max(0, box.x - pad)
    y0 = max(0, box.y - pad)
    x1 = min(max_x, box.x + box.w + pad)
    y1 = min(max_y, box.y + box.h + pad)
    return CropBox(x0, y0, max(1, x1 - x0), max(1, y1 - y0))
```

def _run(coro):
    loop = asyncio.new_event_loop()
    try:
        return loop.run_until_complete(coro)
    finally:
        loop.close()


def _png_b64(img):
    buf = io.BytesIO()
    img.save(buf, format="PNG")
    return base64.b64encode(buf.getvalue()).decode()


class RecordingEditProvider:
    capabilities = frozenset({"raw_rgba"})

    def __init__(self):
        self.calls = []
        self.model = "gpt-image-2"

    def edit_capabilities(self):
        from vulca.providers.base import ImageEditCapabilities
        return ImageEditCapabilities(
            supports_edits=True,
            requires_mask_for_edits=True,
            supports_masked_edits=True,
            supports_unmasked_edits=False,
            supports_quality=True,
        )

    async def inpaint_with_mask(self, *, image_path, mask_path, prompt, **kwargs):
        with Image.open(image_path) as probe:
            size = probe.size
        self.calls.append({"image_path": image_path, "mask_path": mask_path, "size": size, "prompt": prompt})
        return type("Result", (), {
            "image_b64": _png_b64(Image.new("RGB", size, (240, 240, 210))),
            "mime": "image/png",
            "metadata": {},

    async def generate(self, prompt, **kwargs):
        raise AssertionError("sparse crop route must not call maskless generate")

def _crop_rgba_and_alpha(src_rgba, crop_box):
    box = (crop_box.x, crop_box.y, crop_box.x + crop_box.w, crop_box.y + crop_box.h)
    crop = src_rgba.crop(box).convert("RGBA")
    alpha = crop.split()[-1]
    return crop, alpha


def _paste_crop_preserving_alpha(canvas, crop_rgba, crop_alpha, crop_box):
    crop_rgba = crop_rgba.convert("RGBA")
    if crop_rgba.size != crop_alpha.size:
        crop_rgba = crop_rgba.resize(crop_alpha.size)
    crop_rgba.putalpha(crop_alpha)
    canvas.alpha_composite(crop_rgba, dest=(crop_box.x, crop_box.y))

```python
from __future__ import annotations

from dataclasses import dataclass

import numpy as np
from PIL import Image


@dataclass(frozen=True)
class RedrawQualityReport:
    passed: bool
    failures: tuple[str, ...]
    metrics: dict[str, float]


def evaluate_redraw_quality(source_rgba: Image.Image, output_rgba: Image.Image) -> RedrawQualityReport:
    src = np.array(source_rgba.convert("RGBA"))
    out = np.array(output_rgba.convert("RGBA"))
    failures: list[str] = []

    src_a = src[..., 3] > 0
    out_a = out[..., 3] > 0
    src_bbox_area = _bbox_area(src_a)
    out_bbox_area = _bbox_area(out_a)
    bbox_ratio = out_bbox_area / src_bbox_area if src_bbox_area else 0.0
    if bbox_ratio > 2.5:
        failures.append("alpha_bbox_expanded")

    out_rgb = out[..., :3]
    white = (out_rgb[..., 0] > 245) & (out_rgb[..., 1] > 245) & (out_rgb[..., 2] > 245) & out_a
    white_pct = 100.0 * float(white.sum()) / float(max(1, out_a.sum()))
    if white_pct > 85.0 and out_a.sum() > 1000:
        failures.append("large_white_component")

    return RedrawQualityReport(
        passed=not failures,
        failures=tuple(failures),
        metrics={"bbox_ratio": float(bbox_ratio), "white_pct": float(white_pct)},
    )


def _bbox_area(mask: np.ndarray) -> int:
    if not mask.any():
        return 0
    ys, xs = np.where(mask)
    return int((xs.max() - xs.min() + 1) * (ys.max() - ys.min() + 1))
```
