\section*{Research \& Open-Source Projects}

\cventry{2026 – present}
  {Vulca — Agent-Native Image SDK for Claude Code}
  {Sole architect, maintainer, and first author of the underlying research}
  {%
  Productionised the L1–L5 cultural evaluation framework from the EMNLP 2025 paper into
  an agent-native image SDK targeting Anthropic's MCP (Model Context Protocol) ecosystem.
  \begin{itemize}
    \item Designed and shipped \textbf{21 MCP tools} and \textbf{3 end-to-end skills}
    (\texttt{/decompose}, \texttt{/visual-brainstorm}, \texttt{/visual-spec}).
    \item Authored \textbf{13 cultural/design tradition YAML definitions}
    (Chinese Gongbi/Xieyi, Japanese traditional, Islamic geometric, Western academic,
    South Asian, African, watercolour, photography, brand/UI-UX design, contemporary art,
    plus a generic baseline).
    \item Verified an end-to-end local stack on Apple Silicon (ComfyUI + Ollama + EVF-SAM),
    ensuring data-never-leaves-device compliance.
    \item Released v0.14.0 $\to$ v0.17.7 on PyPI (Apache 2.0) — 11+ versioned releases in 14 days,
    with a fully standardised \emph{brainstorm $\to$ spec $\to$ plan $\to$ ship-gate} engineering workflow
    open-sourced alongside the code.
    \item Integrated into the Claude Code plugin marketplace as \texttt{vulca-org/vulca-plugin}.
    \item Repository: \href{https://github.com/vulca-org/vulca}{github.com/vulca-org/vulca}.
  \end{itemize}}
